#include <stdio.h>
#include "soma.h"

int main() {
    char nome[50];
    int num;


    printf("\nInsira seu nome: \n");
    scanf("%s", nome);
    while (!(nome[0] >= 'A' && nome[0] <= 'Z')) {
        printf("Por favor, insira um nome que comece com uma letra maiúscula: \n");
        scanf("%s", nome);
    }

    do {
        printf("\nO que voce gostaria de ver hoje? \n");
        printf("Clique em qualquer numero de 1 a 6: \n");
        printf("\n1- Soma \n2- Subtração \n3- Divisão \n4- Fatorial \n5- Fibonacci \n0- Encerramento\n");

        scanf("%d", &num);

        switch (num) {
            case 1:
                printf("Executando a Soma: \n");
                soma();
                break;

            case 2:
                printf("Executando a Subtração: \n");
                sub();
                break;

            case 3:
                printf("Executando a Divisão: \n");
                divi();
                break;

            case 4:
                printf("Executando o Fatorial: \n");
                fat();
                break;

            case 5:
                printf("Executando a Fibbnacci: \n");
                fibo();
                break;
                
            case 6:
                printf("Executando o sortei de ateh 21: \n");
                sorteio();
                break;

            case 0:
                printf("Saindo...\n");
                break;

            default:
                printf("Digite algo válido.\n");
        }
    } while (num != 0);

    return 0;
}
